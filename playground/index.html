Dead-simple Integration

Compose complex ML transformations using the intuitive Pipeline API.

import { WarpPipeline, IntentAdapter, QuantizationAdapter } from '@warpvector/core';

// 1. Initialize Pipeline
const pipeline = new WarpPipeline(1536)
  .add(new IntentAdapter({
    'beginner': matrixA,
    'expert': matrixB
  }))
  .setFinalStage(new QuantizationAdapter('int8'));

// 2. Warp embeddings at runtime
const result = await pipeline.run({
  vector: queryEmbedding,
  context: { intent: 'expert' }
});

console.log(result.vector); // Int8Array (highly compressed, intent-focused)
